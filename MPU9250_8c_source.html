39 void MPU9250_Start(void) {
40  // This function starts the MPU9250.
41 
42  // Check if the I2C component has already been started,
43  // otherwise start it.
44  if (!I2C_MPU9250_Master_initVar) {
45  I2C_MPU9250_Master_Start();
46  CyDelay(10);
47  }
48 
49  // Wake up MPU9250
50  MPU9250_WakeUp();
51 
52  // Set up default accelerometer full scale range
53  MPU9250_SetAccFS(MPU9250_Acc_FS_2g);
54 
55  // Set up defaul gyroscope full scale range
56  MPU9250_SetGyroFS(MPU9250_Gyro_FS_250);
57 
58  // Set up sample rate divider
59  MPU9250_SetSampleRateDivider(4); // From 1kHz to 200 Hz sampling
60 
61  // Set up gyroscope, temperature digital low pass filter
62  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_CONFIG_REG, 0x03);
63 
64  // Set up accelerometer digital low pass filter
65  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_2_REG, 0x03);
66 
67  // Configure interrupt

69  MPU9250_SetInterruptPushPull();

116 void MPU9250_ReadAcc(int16_t* acc) {
117  // We can read 6 consecutive bytes since the accelerometer registers are in order
118 
119  uint8_t temp[6]; // Temp variable to store the data
120 
121  // Read data via I2C
122  MPU9250_I2C_ReadMulti(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_XOUT_H_REG, temp, 6);
123  acc[0] = (temp[0] << 8) | (temp[1] & 0xFF);
124  acc[1] = (temp[2] << 8) | (temp[3] & 0xFF);
125  acc[2] = (temp[4] << 8) | (temp[5] & 0xFF);
126 }

212 void MPU9250_SelfTest(float* deviation) {
213  // Perform self test of accelerometer and gyroscope according to the
214  // procedure described in the application note MPU-9250 Accelerometer, Gyroscope and
215  // Compass Self-Test Implementation.
216 
217  MPU9250_Acc_FS Old_Acc_FS; // Old value of accelerometer full scale range
218  MPU9250_Gyro_FS Old_Gyro_FS; // Old value of accelerometer full scale range
219  int16_t ST_Acc[3] = {0,0,0}; // Accelerometer values with self test enabled
220  int16_t Acc[3] = {0,0,0}; // Accelerometer average valus without self test enabled
221  int16_t ST_Gyro[3] = {0,0,0}; // Gyroscope values with self test enabled
222  int16_t Gyro[3] = {0,0,0}; // Gyroscope values without self test enabled
223  int16_t ST_Response[6]; // Self test response on acc and gyro 3 axis
224 
225  // Get current accelerometer full scale range
226  Old_Acc_FS = MPU9250_GetAccFS();
227  // Get current gyroscope full scale range
228  Old_Gyro_FS = MPU9250_GetGyroFS();
229 
230  // Set gyroscope full scale range to 250dps
231  MPU9250_SetGyroFS(MPU9250_Gyro_FS_250);
232  // Set accelerometer full scale range to 2g
233  MPU9250_SetAccFS(MPU9250_Acc_FS_2g);
234 
235  // Write 0 to sample rate divider register -- no additional divider
236  MPU9250_SetSampleRateDivider(0x00);
237 
238  // Set gyroscope and accelerometer DLPF configuration to 1kHz Fs, 92 Hz bandwidth
239  // First set up configuration register so that DLPF CFG is set to 2
240  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_CONFIG_REG, 0x02);
241  // Then write 00 in FChoice_b of GYRO config register
242  uint8_t temp = MPU9250_I2C_Read(MPU9250_I2C_ADDRESS, MPU9250_GYRO_CONFIG_REG);
243  temp &= ~ 0x02; // Clear bits [1:0]
244  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_GYRO_CONFIG_REG, temp);
245  // Set accelerometer DLPF configuration
246  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_REG, 0x02);
247  // Get 200 readings
248  int16_t Acc_Temp[3];
249  int16_t Gyro_Temp[3];
250 
251  for (int reading = 0; reading < 200; reading++) {
252  MPU9250_ReadAcc(Acc_Temp);
253  Acc[0] += Acc_Temp[0];
254  Acc[1] += Acc_Temp[1];
255  Acc[2] += Acc_Temp[2];
256  MPU9250_ReadGyro(Gyro_Temp);
257  Gyro[0] += Gyro_Temp[0];
258  Gyro[1] += Gyro_Temp[1];
259  Gyro[2] += Gyro_Temp[2];
260  }
261  // .. and average them
262  for (int i = 0; i < 3; i++) {
263  Acc[i] /= 200;
264  Gyro[i] /= 200;
265  }
266  // Enable self test gyroscope
267  temp = MPU9250_I2C_Read(MPU9250_I2C_ADDRESS, MPU9250_GYRO_CONFIG_REG);
268  // Then, set bits [7,6,5]
269  temp |= 0b11100000;
270  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_GYRO_CONFIG_REG, temp);
271  // Enable self test accelerometer
272  temp = MPU9250_I2C_Read(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_REG);
273  // Then, set bits [7,6,5]
274  temp |= 0b11100000;
275  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_REG, temp);
276 
277  // Wait 20 ms so that everything is stable
278  CyDelay(20);
279 
280  for (int reading = 0; reading < 200; reading++) {
281  MPU9250_ReadAcc(Acc_Temp);
282  ST_Acc[0] += Acc_Temp[0];
283  ST_Acc[1] += Acc_Temp[1];
284  ST_Acc[2] += Acc_Temp[2];
285  MPU9250_ReadGyro(Gyro_Temp);
286  ST_Gyro[0] += Gyro_Temp[0];
287  ST_Gyro[1] += Gyro_Temp[1];
288  ST_Gyro[2] += Gyro_Temp[2];
289  }
290  // .. and average them
291  for (int i = 0; i < 3; i++) {
292  ST_Acc[i] /= 200;
293  ST_Gyro[i] /= 200;
294  }
295  // Disable self test gyroscope -- Read config register
296  temp = MPU9250_I2C_Read(MPU9250_I2C_ADDRESS, MPU9250_GYRO_CONFIG_REG);
297  // Clear bits [7,6,5]
298  temp &= ~0b11100000;
299  // Write new value to the register
300  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_GYRO_CONFIG_REG, temp);
301  // Disbale self test accelerometer -- Read config register
302  temp = MPU9250_I2C_Read(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_REG);
303  // Clear bits [7,6,5]
304  temp &= ~0b11100000;
305  // Write new vale to the register
306  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_REG, temp);
307 
308  // Wait 20 ms so that everything is stable
309  CyDelay(20);
310 
311  // Restore full scale range values
312  MPU9250_SetAccFS(Old_Acc_FS);
313  MPU9250_SetGyroFS(Old_Gyro_FS);
314 
315  // Compute self test responses
316  ST_Response[0] = ST_Acc[0] - Acc[0];
317  ST_Response[1] = ST_Acc[1] - Acc[1];
318  ST_Response[2] = ST_Acc[2] - Acc[2];
319  ST_Response[3] = ST_Gyro[0] - Gyro[0];
320  ST_Response[4] = ST_Gyro[1] - Gyro[1];
321  ST_Response[5] = ST_Gyro[2] - Gyro[2];
322 
323  // Get values stored in the device for self test
324  int16_t ST_AccStored[3];
325  int16_t ST_GyroStored[3];
326 
327  MPU9250_ReadSelfTestAcc(ST_AccStored);
328  MPU9250_ReadSelfTestGyro(ST_GyroStored);
329 
330  // Compute factory trim
331  float Factory_Trim[6];
332  Factory_Trim[0] = (float) ( ( 2620 / 1 << MPU9250_Acc_FS_2g) * pow( 1.01, ST_AccStored[0] - 1.0) );
333  Factory_Trim[1] = (float) ( ( 2620 / 1 << MPU9250_Acc_FS_2g) * pow( 1.01, ST_AccStored[1] - 1.0) );
334  Factory_Trim[2] = (float) ( ( 2620 / 1 << MPU9250_Acc_FS_2g) * pow( 1.01, ST_AccStored[2] - 1.0) );
335  Factory_Trim[3] = (float) ( ( 2620 / 1 << MPU9250_Acc_FS_2g) * pow( 1.01, ST_GyroStored[0] - 1.0) );
336  Factory_Trim[4] = (float) ( ( 2620 / 1 << MPU9250_Acc_FS_2g) * pow( 1.01, ST_GyroStored[1] - 1.0) );
337  Factory_Trim[5] = (float) ( ( 2620 / 1 << MPU9250_Acc_FS_2g) * pow( 1.01, ST_GyroStored[2] - 1.0) );
338 
339  // Compute deviation
340  for (int i = 0; i < 3; i++) {
341  deviation[i] = 100.0 * ((float) (ST_Response[i] - Acc[i])) / Factory_Trim[i] - 100;
342  deviation[i+3] = 100.0 * ((float) (ST_Response[i] - Gyro[i+3])) / Factory_Trim[i] - 100;
343  }
344 }

346 void MPU9250_SetAccFS(MPU9250_Acc_FS fs) {
347  // Write the new full scale value in the acc conf register
348 
349  // We need to first read the current bits of the register
350  uint8_t temp = MPU9250_I2C_Read(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_REG);
351  // Then, we clear bits [4:3]
352  temp &= ~MPU9250_ACC_FS_MASK;
353  // Lastly, we write the new byte to the register
354  MPU9250_I2C_Write(MPU9250_I2C_ADDRESS, MPU9250_ACCEL_CONFIG_REG, temp | ( fs << 3));
355 
356  // We also need to update the scaling factor
357  switch(fs) {
358  case MPU9250_Acc_FS_2g:
359  acc_scale = MPU9250_G * 2.0f/pow(2,16)/2;
360  break;
361  case MPU9250_Acc_FS_4g:
362  acc_scale = MPU9250_G * 4.0f/pow(2,16)/2;
363  break;
364  case MPU9250_Acc_FS_8g:
365  acc_scale = MPU9250_G * 8.0f/pow(2,16)/2;
366  break;
367  case MPU9250_Acc_FS_16g:
368  acc_scale = MPU9250_G * 16.0f/pow(2,16)/2;
369  break;
370 
371  }
372 }
