14  /*
15  Standard I2C single byte read protocol
16  - Send start signal requesting write operation
17  - Write byte with register address
18  - Send stop signal
19  - Send start signal requesting read operation
20  - Read byte without acknowledgement
21  - Send stop
22  */
23 
24  I2C_MPU9250_Master_MasterSendStart(address, I2C_MPU9250_Master_WRITE_XFER_MODE);
25  I2C_MPU9250_Master_MasterWriteByte(reg);

33 void MPU9250_I2C_ReadMulti(uint8_t address, uint8_t reg, uint8_t* data, uint16_t count) {
34  /*
35  Standard I2C multi bytes read protocol
36  - Send start signal requesting write operation
37  - Write byte with first register address
38  - Send restart signal
39  - While loop with read byte and acknowledgement
40  - Last byte read without acknowledgement
41  - Send stop
42  */
43  I2C_MPU9250_Master_MasterSendStart(address, I2C_MPU9250_Master_WRITE_XFER_MODE);
44  I2C_MPU9250_Master_MasterWriteByte(reg);
45  I2C_MPU9250_Master_MasterSendRestart(address,I2C_MPU9250_Master_READ_XFER_MODE);
46  while (count--) {
47  if (!count) {
48  /* Last byte */
49  *data++ = I2C_MPU9250_Master_MasterReadByte(I2C_MPU9250_Master_NAK_DATA);
50  } else {
51  *data++ = I2C_MPU9250_Master_MasterReadByte(I2C_MPU9250_Master_ACK_DATA);
52  }
53  }
54  I2C_MPU9250_Master_MasterSendStop();
55 }
